/// IF VERTEXMOTION

		float w = 0;

	/// ELIF VERTEXMOTION_NORMAL
		vertExmotionPos.xyz = VertExmotionWorldPosASE(vertExmotionPos.xyz, vxData, normal, tangent);
	///
		vertex.xyz = mul(unity_WorldToObject, vertExmotionPos).xyz;
